    public void xwins(int a,int b,int c){

public void  actionPerformed(ActionEvent e)
    {
        for(int i=0;i<9;i++){
            if(e.getSource()==b1[i]){
                if(player1_turn){
                    if(b1[i].getText()==""){
                        b1[i].setText("X");
                        player1_turn=false;
                        l1.setText("O turn");
                        check();
                    }
                }
                else{
                    if(b1[i].getText()==""){
                        b1[i].setText("O");
                        player1_turn=true;
                        l1.setText("X turn");
                         check();
                    }
                    
                }
            }
        }
    
    }
